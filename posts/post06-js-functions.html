JavaScript Functions

Functions are used for perform a particular task.

A function exemple to observe the syntax: 

          
            function calculateAge(name, birthYear) {

              let date = new Date();
              let currentYear = date.getFullYear(); 
              let age = currentYear - birthYear;
            
              let display = name + " is " + age + " years old!";
                        
              return display; // EXIT here !

            }

A function that returns day of the week based on a number

          
            function dayOfTheWeek(index) {
        
              let days = ["Monday", "Tuesday", "Wednesday", "Thursday", "Friday", 
              "Saturday", "Sunday"];
            
              return days[index];
            
            }

A function which produces a greeting message

          
            function sayHello(greeting, input) {

              console.log(greeting + " " + input + " !");
            
            }
            sayHello("Hello", "Bob"); // Returns "Hello Bob !"

How many seconds are there in "minutes" minutes?

          
            function minToSec(minutes) {

              console.log(minutes + " minutes = " + (minutes * 60) + " seconds" );
            
            }
            minToSec(60); // Returns "60 minutes = 3600 seconds"
